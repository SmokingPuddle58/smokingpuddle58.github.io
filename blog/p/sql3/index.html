Database with SQLite3 in Python

Mar 13, 2024
3 minute read

Import necessary libraries

1

import sqlite3

Connect to database

1
2
3
4
5
6
7

# The database that you want to access
# If the database does not exist, it will be created
connection = sqlite3.connect('mydatabase.db') 

# This will create a new database in the memory only.
# The database vanishes when the program ends
connection = sqlite3.connect(':memory:')

# Create a table

        var3 DATATYPE
    )""")

Datatype

  • NULL
    • Indicating existence. NULL if not exist.
  • INTEGER
    • Integer value.
  • REAL
    • Float value.
  • TEXT
    • Text value.
  • BLOG:
    • Binary large object.

WHERE clause

1
2
3
4
5

# To select data that with var1 is 'value1'
cursor.execute("SELECT * FROM tablename WHERE var1 = 'value1'")

# To select data that with var1 begins with 'val'. % is a wildcard character.
cursor.execute("SELECT * FROM tablename WHERE var1 LIKE 'val%'")

Update records

 2
 3
 4
 5
 6
 7
 8
 9

11

# Update all records with var2 = 'value2' to var1 = 'new_value'
cursor.execute("""
    UPDATE tablename SET var1 = 'new_value'
    WHERE var2 = 'value2'
""")

# Update record with rowid 1 to var1 = 'new_value'
cursor.execute("""
    UPDATE tablename SET var1 = 'new_value'
    WHERE rowid = 1
""")

Ordering

1
2

# Order by rowid in descending order

AND/OR

1
2

# To select data that with var1 begins with 'val' and rowid is 3.
cursor.execute("SELECT * FROM tablename WHERE var1 LIKE 'val%' AND rowid = 3")

LIMIT

1
2

# To select data that with var1 begins with 'val' and rowid is 3. Limit to 2 results
cursor.execute("SELECT * FROM tablename WHERE var1 LIKE 'val%' AND rowid = 3 LIMIT 2")
